        buildGlowGradient: function(canvasCtx, startColorHex, endColorHex, gradientHeight = 350) {
            const gradient = canvasCtx.createLinearGradient(0, 0, 0, gradientHeight);
            gradient.addColorStop(0, GraphicalTools.hexToRgba(startColorHex, 0.65));
            gradient.addColorStop(1, GraphicalTools.hexToRgba(endColorHex || startColorHex, 0.02));
            return gradient;
        },
        
        /**
         * Crea el gradiente específico para barras de aprobación/riesgo
         */
        buildStateBarGradient: function(canvasCtx, isSuccessState) {
            const gradient = canvasCtx.createLinearGradient(0, 0, 0, 250);
            if (isSuccessState) {
                gradient.addColorStop(0, SYS_CONFIG.PALETTE.successLight);
                gradient.addColorStop(1, SYS_CONFIG.PALETTE.success);
            } else {
                gradient.addColorStop(0, SYS_CONFIG.PALETTE.dangerLight);
                gradient.addColorStop(1, SYS_CONFIG.PALETTE.danger);
            }
            return gradient;
        }
    };

    const InteractionEngine = {
        
        initializeScrollObservers: function() {
            // Observador para animar elementos y cargar gráficos cuando entran en pantalla
            const scrollObserver = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        
                        // Disparar animación CSS
                        entry.target.classList.add('is-visible');
                        
                        // Buscar si hay gráficos Lazy en este contenedor y renderizarlos
                        const lazyCanvasList = entry.target.querySelectorAll('.lazy-chart');
                        lazyCanvasList.forEach(canvas => {
                            if (canvas.dataset.chartType === 'doughnut') {
                                VisualDataEngine.renderDonutChart(canvas.id, canvas.dataset.chartData);
                            }
                            if (canvas.dataset.chartType === 'bar') {
                                VisualDataEngine.renderBarChart(canvas.id, canvas.dataset.labels, canvas.dataset.values);
                            }
                        });

                        // Buscar si es el contenedor del gráfico global
                        if (entry.target.querySelector('.lazy-chart-global')) {
                            VisualDataEngine.renderGlobalMatrixLineChart();
                        }
                        
                        // Dejamos de observar una vez que ya apareció y cargó
                        scrollObserver.unobserve(entry.target);
                    }
                });
            }, { 
                threshold: 0.1, 
                rootMargin: "0px 0px 80px 0px" // Pre-carga 80px antes de que entre al viewport
            });

            // Asignar delays escalonados (Stagger) para un efecto cascada premium
            document.querySelectorAll('.observe-stagger').forEach((element, index) => {
                element.classList.add('stagger-item');
                // Efecto cascada reiniciable por filas (módulo 4)
                element.style.transitionDelay = `${(index % 4) * 0.1}s`;
                scrollObserver.observe(element);
            });
        },

        initializeAccordionEvents: function() {
            // Escuchar cuando Bootstrap abre un acordeón
            document.querySelectorAll('.collapse').forEach(element => {
                element.addEventListener('shown.bs.collapse', function(event) {
                    // Pasar el target al motor visual para que pinte los gráficos lazy
                    VisualDataEngine.lazyRenderAccordionInnerCharts(event.target);
                });
            });
        }
    };

        // Petición asíncrona robusta al endpoint de Django (Agent IA)
        dispatchNetworkRequest: async function(actionType, userQuery, extraParams = {}) {
            try {
                // Incorporar el token CSRF requerido por Django
                const csrfTokenElement = document.querySelector('[name=csrfmiddlewaretoken]');
                // Fallback directo a la template tag de django
                const rawToken = '{{ csrf_token }}'; 
                
                const requestPayload = {
                    action_type: actionType,
                    query: userQuery,
                    params: extraParams,
                    historial: sessionNeuralMemory // Envío de contexto conversacional
                };

                const networkResponse = await fetch("{% url 'api_ai_agent' %}", {
                    method: 'POST',
                    headers: { 
                        'Content-Type': 'application/json', 
                        'X-CSRFToken': rawToken 
                    },
                    body: JSON.stringify(requestPayload)
                });
                
                return await networkResponse.json();
            } catch (networkError) {
                console.error("AI Node Uplink Disconnected:", networkError);
                return { 
                    success: false, 
                    content: "Falla crítica en la comunicación con el núcleo del servidor. Verifique conectividad local." 
                };
            }
        }

        toggleSimulatorInstance: function() {
            const wrapperNode = document.getElementById('wrapperSimulador');
            const iframeNode = document.getElementById('iframeSimulador');
            const loaderNode = document.getElementById('simLoader');
            
            if (wrapperNode.classList.contains('d-none')) {

                wrapperNode.classList.remove('d-none');
                wrapperNode.classList.add('animate__animated', 'animate__zoomIn');

                // Inyectar URL al iframe de manera asíncrona para no bloquear el hilo de UI

                    iframeNode.src = SYS_CONFIG.SIMULATOR_SECURE_URI; 
                    
                    // Escuchar cuando el Vercel termine de cargar para hacer Fade del loader

                        setTimeout(() => {

                            setTimeout(() => {
                                loaderNode.classList.add('d-none');
                                loaderNode.classList.remove('animate__animated', 'animate__fadeOut');
                                iframeNode.classList.add('loaded'); // Activa opacity:1 en CSS
                            }, 500);
                        }, 800); // Pequeño delay extra de UX
                    };
                }, 300); 
                
                // Hacer scroll fluido hasta la consola
                setTimeout(() => {
                    wrapperNode.scrollIntoView({ behavior: 'smooth', block: 'center' });
                }, 100);
                
            } else {
                // FASE DE DESTRUCCIÓN: Animación de salida y limpieza de memoria
                wrapperNode.classList.replace('animate__zoomIn', 'animate__zoomOut');
                
                setTimeout(() => {
                    wrapperNode.classList.add('d-none');
                    wrapperNode.classList.remove('animate__animated', 'animate__zoomOut');
                    iframeNode.classList.remove('loaded');
                    
                    // HARD GARBAGE COLLECTION: Matar la instancia del iframe removiendo el SRC
                    iframeNode.removeAttribute('src'); 
                    
                }, 500); // Mismo tiempo que la duración de la animación zoomOut
            }
        },

            } else if (iframeElement.msRequestFullscreen) { /* IE11 */

        triggerSystemAI: async function(actionCode, initialPromptText) {

                sessionNeuralMemory = [];
                chatInnerBody.innerHTML = '';
                NeuralTutorEngine.injectChatUI('loading');
            }
            
            bootstrapModal.show();
            
            // Petición bloqueante al backend
            const responseData = await NeuralTutorEngine.dispatchNetworkRequest(actionCode, initialPromptText);
            
            NeuralTutorEngine.destroyLoadingState();

            if (responseData.success) {
                // Guardar en la memoria local el hilo
                if (initialPromptText) {
                    sessionNeuralMemory.push({ role: "user", content: initialPromptText });
                }
                sessionNeuralMemory.push({ role: "assistant", content: responseData.content });
                
                // Pintar la respuesta
                NeuralTutorEngine.injectChatUI('assistant', responseData.content);
            } else {
                // Manejo de excepciones visuales
                NeuralTutorEngine.injectChatUI('assistant', `<span class="text-danger fw-bold"><i class="fas fa-bug me-2"></i>Fallo en subsistema: ${responseData.content}</span>`, true);
            }
        },

         * Controlador de chat de entrada de usuario manual (Input)
         */

            const userString = inputElement.value.trim();

            NeuralTutorEngine.injectChatUI('user', userString);
            inputElement.value = ''; 
            
            // Poner loader IA
            NeuralTutorEngine.injectChatUI('loading');

            // Enviar petición (Sin actionType, se asume continuación de chat libre)
            const responseData = await NeuralTutorEngine.dispatchNetworkRequest(null, userString);
            
            NeuralTutorEngine.destroyLoadingState();

            if (responseData.success) {
                sessionNeuralMemory.push({ role: "user", content: userString });
                sessionNeuralMemory.push({ role: "assistant", content: responseData.content });
                NeuralTutorEngine.injectChatUI('assistant', responseData.content);
            } else {
                NeuralTutorEngine.injectChatUI('assistant', `<span class="text-danger fw-bold"><i class="fas fa-wifi me-2"></i>Avería de transmisión en túnel principal.</span>`, true);
            }
        }
    };
})();
